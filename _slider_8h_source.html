302 template <typename Type>
303 void Slider<Type>::onPointerEvent(DOM::PointerUIEventArgs& e)
304 {
305  if (e.pointer().id() == _primaryPointerId)
306  {
307  if (PointerEventArgs::POINTER_DOWN == e.type() ||
308  PointerEventArgs::POINTER_MOVE == e.type())
309  {
310  // Convert to integer to avoid warnings w/ Point.
311  std::size_t axisIndex = _getActiveAxisIndex();
312 
313  // Local position.
314  float position = screenToLocal(e.pointer().point())[static_cast<int>(axisIndex)];
315 
316  Type valueMin = _value.getMin();
317  Type valueMax = _value.getMax();
318 
319  if (_isInverted)
320  {
321  std::swap(valueMin, valueMax);
322  }
323 
324  float sliderMin = 0;
325  float sliderMax = getSize()[static_cast<int>(axisIndex)];
326 
327  if (DOM::Orientation::VERTICAL == _getEffectiveOrientation())
328  {
329  std::swap(sliderMin, sliderMax);
330  }
331 
332  Type _lerpedValue = Math::lerp(position,
333  sliderMin,
334  sliderMax,
335  valueMin,
336  valueMax);
337 
338 
339  if (DragMode::RELATIVE == _dragMode)
340  {
341  if (PointerEventArgs::POINTER_DOWN == e.type())
342  {
343  // Find the difference between the current value
344  // and the value underneath the pointer. This is
345  // the value offset.
346  _valueOffset = (_value - _lerpedValue);
347  return;
348  }
349  else
350  {
351  // Add the value offset to the calculated value based
352  // on the poiter position.
353  _lerpedValue += _valueOffset;
354  }
355  }
356 
357  _value = Math::clamp(_lerpedValue, valueMin, valueMax);
358 
359  }
360  }
361 }

380 template <typename Type>
381 void Slider<Type>::onDraw() const
382 {
383  Widget::onDraw();
384 
385  auto styles = getStyles();
386 
387  ofFill();
388 
389  if (isPointerDown())
390  {
391  ofSetColor(styles->getColor(Styles::ROLE_FOREGROUND, Styles::STATE_DOWN));
392  }
393  else if (isPointerOver())
394  {
395  ofSetColor(styles->getColor(Styles::ROLE_FOREGROUND, Styles::STATE_OVER));
396  }
397  else
398  {
399  ofSetColor(styles->getColor(Styles::ROLE_FOREGROUND, Styles::STATE_NORMAL));
400  }
401 
402  DOM::Orientation orientation = _orientation;
403 
404  if (DOM::Orientation::DEFAULT == orientation)
405  {
406  orientation = (getWidth() > getHeight()) ? DOM::Orientation::HORIZONTAL : DOM::Orientation::VERTICAL;
407  }
408 
409  Type min = _value.getMin();
410  Type max = _value.getMax();
411 
412  if (_isInverted)
413  {
414  std::swap(min, max);
415  }
416 
417  if (DOM::Orientation::HORIZONTAL == orientation)
418  {
419  float width = Math::lerp(_value, min, max, 0, getWidth(), true);
420  ofDrawRectangle(0, 0, width, getHeight());
421  }
422  else
423  {
424  float height = Math::lerp(_value, min, max, 0, getHeight(), true);
425  ofDrawRectangle(0, getHeight(), getWidth(), - height);
426  }
427 
428  ofNoFill();
429  ofSetColor(styles->getColor(Styles::ROLE_BORDER, Styles::STATE_NORMAL));
430  ofDrawRectangle(0, 0, getWidth(), getHeight());
431 }
